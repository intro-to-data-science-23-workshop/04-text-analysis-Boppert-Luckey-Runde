Exercise 11: We want to create a function that counts the words in a sentence, a paragraph or, for example, a tweet. Think about this: Do you want to include punctuation? And if not, how can we make sure these signs are not counted? Make sure to try out the function with our motto from the beginner exercises.

word_frequency <- function(sentence) {
  sentence <- str_replace_all(sentence, "[[:punct:]]", "") #exclude any kind of punctuation 
  words <- unlist(str_split(sentence, "\\s+"))
  table(words)
}

# Example usage

remove_pii_from_emails <- function(string) {
  if (str_detect(string, valid_email)) {
    return(str_replace(string, " .+@", " user@"))
  }
  return(string)
}

remove_credit_card_numbers <- function(string) {
  str_replace(string, "[0-9]{4}-[0-9]{4}-[0-9]{4}-[0-9]{4}", "CREDIT-CARD")
}
